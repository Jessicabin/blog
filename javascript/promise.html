# 串行期约

使用场景：需要批量进行一组请求，由于后台限制，不可并发执行，只能按顺序一个个执行，同时显示请求进度。

function fetchSequentially(ids) {
   let doneCount = 0

    function fetchOne(val) {
        doneCount++
        console.log(`已完成: ${doneCount}/${ids.length}`)
        return fetch(val) //fetch() 假定一个异步API请求
    }

    //构造期约链
    let p = Promise.resolve(undefined) //以一个立即（以undefined值）兑现的期约开始

    //根据传入的请求参数构建任意长度的期约
    for (let id of ids) {
        p = p.finally(() => fetchOne(id))
    }

    return p.then(res => { })
            .catch(err => { })
            .finally(data => {
                //请求全部结束
            })
    }
}

# 并行期约

Promise.all()

promise.all 接收一组 promise，并发执行所有期约，当全部兑现（fulfilled）返回结果数组，如果其中一个 promise 失败，立即返回结果。

const p1 = await fetch(1),
  p2 = await fetch(1);

const [value1, value2] = await Promise.all([p1, p2]);
