# vuex 批量引入store文件

由于store分模块写，所以在index文件中引入所有

文件路径

store
|—— modules
|   |__ xxxx.js
|—— index.js

参考：Glob 导入open in new window

import { createStore, createLogger } from 'vuex'

const debug = process.env.NODE_ENV !== 'production'

const files = import.meta.globEager('./modules/*.js')

const modules = {}
for (const key in files) {
  modules[key.replace(/(\.\/modules\/|\.js)/g, '')] = files[key].default
}

Object.keys(modules).forEach(item => {
  modules[item]['namespaced'] = true
})

export default createStore({
  modules,
  strict: debug,
  plugins: debug ? [createLogger()] : []
})

在vue中这样引入（vue2写法）

import { mapState, mapGetters, mapActions } from 'vuex'

computed: {
  ...mapGetters('storeName', ['variableName']),
  ...mapState('storeName', ['variableName']),
},
methods: {
  ...mapActions('storeName', ['functionName']),
}
