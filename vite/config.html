# 一般配置

import { defineConfig } from 'vite'
import { createVuePlugin } from 'vite-plugin-vue2' //vue2
// import vue from '@vitejs/plugin-vue' //vue3
import path from 'path'

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [createVuePlugin(/* options */)], //vue2
  // plugins: [vue()], //vue3
  server: {
    port: 8081,
    host: '0.0.0.0',
    proxy: {
      '/server': {
        target: 'http://localhost:8100',
        changeOrigin: true,
        rewrite: (_path) => _path.replace(/^\/server/, ''),
      },
    },
  },
  resolve: {
    alias: [
      { find: '@', replacement: path.resolve(__dirname, './src') },
      { find: 'common', replacement: path.resolve(__dirname, './src/common') },
      { find: 'api', replacement: path.resolve(__dirname, './src/api') },
      { find: 'views', replacement: path.resolve(__dirname, './src/views') },
    ],
    extensions: [
      '.mjs', '.js', '.ts', '.jsx', '.tsx', '.json', '.vue',
    ],
  },
  css: {
    preprocessorOptions: {
      scss: {
        additionalData: '@import "@/styles/variables.scss";'
      }
    }
  },
  build: {
    rollupOptions: {
      output: {
        //打包优化，分块打包，减小index文件体积
        manualChunks: {
          axios: ['axios'],
          moment: ['moment'],
          vue: ['vue', 'vue-router', 'vuex'],
          element: ['element-ui'],
          echarts: ['echarts'],
        }
      }
    }
  }
})
