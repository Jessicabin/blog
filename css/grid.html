关键代码

grid 布局样式，外层样式需限制宽高

display: grid;
justify-content: space-between;
align-content: space-between;

行列样式根据输入数据,动态计算每个 div 长宽

grid-template-columns: repeat(
  ${column},
  calc((100% - $ {(column - 1) * margin}px) / ${column})
);
grid-template-rows: repeat(
  ${row},
  calc((100% - $ {(row - 1) * margin}px) / ${row})
);

完整代码

<div class="banner">
  <div class="b-item">
    列：
    <el-input v-model="column" placeholder="请输入列" />
  </div>
  <div class="b-item">
    行：
    <el-input v-model="row" placeholder="请输入行" />
  </div>
</div>

<div class="grid-wrapper" :style="getGridStyle">
  <div v-for="i in total" :key="i" class="item">{{ i }}</div>
</div>

<script>
import { ElInput } from 'element-plus'
import { ref, computed } from "vue"

export default {
  components: { ElInput },
  setup(){
    const row = ref(3),
          column = ref(3),
          margin = ref(10)

    const getGridStyle = computed(()=>{
      return `
      grid-template-columns: repeat(${column.value}, calc((100% - ${(column.value - 1) * margin.value}px) / ${column.value}));
      grid-template-rows: repeat(${row.value}, calc((100% - ${(row.value - 1) * margin.value}px) / ${row.value}));`
    })
    const total = computed(()=> row.value * column.value)
    return {
      column,
      row,
      margin,
      total,
      getGridStyle
    }
  },
}
</script>
