# vuex 应用

在项目中往往有些数据、方法是需要多处调用的，如字典数据，为了减少资源开销，就需要用到状态管理 vuex。

# 目录结构

为了方便管理，不同模块不同文件，在 index 中统一引入。

store
├── index.js
└── modules
    ├── xxx.js
    └── xxx.js

# 导入所有模块

index.js

自动加入 namespaced:true，用于解决 vuex 命名冲突。

vue2 写法

import Vue from "vue";
import Vuex from "vuex";

Vue.use(Vuex);

//批量引入modules文件

//webpack写法
const files = require.context("./modules", false, /\.js$/);

//vite写法
const files = import.meta.globEager("./modules/*.js");

const modules = {};

files.keys().forEach((key) => {
  modules[key.replace(/(\.\/|\.js)/g, "")] = files(key).default;
});

Object.keys(modules).forEach((key) => {
  modules[key]["namespaced"] = true;
});

const store = new Vuex.Store({
  modules,
});
export default store;

vue3+vite 写法

import { createStore, createLogger } from "vuex";

const debug = process.env.NODE_ENV !== "production";

//批量引入store文件
const files = import.meta.globEager("./modules/*.js");
const modules = {};
for (const key in files) {
  modules[key.replace(/(\.\/modules\/|\.js)/g, "")] = files[key].default;
}

Object.keys(modules).forEach((item) => {
  modules[item]["namespaced"] = true;
});

export default createStore({
  modules,
  strict: debug,
  plugins: debug ? [createLogger()] : [],
});

# 使用

vue2 写法

import { mapState, mapActions, mapGetters } from "vuex";

export default {
  methods: {
    ...mapActions("moduleName", ["actionName"]),
  },
  computed: {
    ...mapGetters("moduleName", ["getterName"]),
    ...mapState("moduleName", ["stateName"]),
  },
};
