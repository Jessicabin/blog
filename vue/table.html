# 基于element-ui的table组件封装

在管理后台中经常需要展示表格，如果每次都在vue页面中复制粘贴element-ui代码，会造成过多重复且无意义的代码，如果封装一个table组件，通过传入配置项即可渲染出列表，可以降低页面重复代码，提高代码可读性以及易于修改，利于协同开发。

# 使用示例

<c-table :table-data="list" 
          :headers="headers" 
          @sortChange="handleSortChange" 
          @selectionChange="handleSelectionChange" 
          @operBtnClick="handleOperBtnClick" />

this.headers = [
  { name: 'name', label: '角色名称', sortable: 'custom', width: 200 },
  { name: 'remark', label: '描述' },
  { name: 'createdTime', label: '创建时间', sortable: 'custom', align: 'center', width: 200 },
  { name: 'operation', label: '操作', align: 'center', width: 80, type: 'operation', config: {} }
]

headers即为列表渲染配置，使用这种低代码方式开发，利于工作团队协同开发，制定统一标准，如有样式等修改，可以统一修改，提高工作效率

# 封装table组件

<el-table ref="table" 
          v-loading="loading" 
          class="base-table" 
          highlight-current-row 
          border 
          stripe 
          :data="tableData"
          @sort-change="handleSortChange" 
          @selection-change="handleSelectionChange">
  <el-table-column v-if="hasSelection" width="50" type="selection" :selectable="isSelected" />

  <el-table-column v-for="header in headers.filter(header => !header.isHidden)"
                    :key="header.name" 
                    :label="header.label" 
                    :prop="header.name" 
                    :width="header.width"
                    :min-width="header.minwidth"
                    :fixed="header.fixed" 
                    :sortable="header.sortable" 
                    :align="header.align"
                    show-overflow-tooltip>
    <template #default="{row}">
      <slot v-if="header.type === 'slot'" :name="header.name" :row="row" />

      <item-render :header="header" 
                    :row="row" 
                    @operBtnClick="handleOperBtnClick" 
                    @rowButtonClick="handleRowButtonClick" />
    </template>
  </el-table-column>

  <template #empty>
    <div v-if="!loading && !tableData.length">
      暂无数据
    </div>
  </template>
</el-table>

import ItemRender from './c-table-item-render'

export default {
  name: 'BaseTable',
  components: {
    ItemRender
  },
  props: {
    tableData: {
      type: Array,
      default: function() {
        return []
      }
    },
    loading: { // 数据加载
      type: Boolean,
      default: false
    },
    headers: {
      type: Array,
      default: function() {
        return []
      }
    },
    hasSelection: {
      type: Boolean,
      default: false
    },
    isSelected: {
      type: Function,
      default: function() {
        return true
      }
    },
  },
  methods: {
    handleSortChange(column, prop, order) {
      this.$emit('sortChange', column, prop, order)
    },
    handleSelectionChange(selection) {
      this.$emit('selectionChange', selection)
    },
    handleOperBtnClick(...args) {
      this.$emit('operBtnClick', ...args)
    },
    handleRowButtonClick(...args) {
      this.$emit('rowButtonClick', ...args)
    }
  },
}

# 表格项渲染

常用table组件有文本展示、tag、字典匹配，使用渲染组件方式有利于减少资源开销。

根据type，渲染对应的组件，避免写许多v-if。

参考：渲染函数open in new window

table-item-render.js

import OperationButton from '@/components/operation-button/operation-button'
import ctag from '../c-tableitems/cTag'

/**
 * 用render方式渲染列表
 * type: 列表项类型
 * {
 *    btnOper：操作按钮, 
 *    iconSvg：svg icon，
 *    time:时间格式化，
 *    map:根据字段映射
 * },
 * config: 列表项其他配置参数-例：map类型需要map映射表，
 */

export default {
  props: {
    header: {},
    row: {},
  },
  methods: {
    getRowColor(config, row) {
      if (config && config.color && config.colorKey) {
        return `color:${config.color[row[config.colorKey]]};`
      }
    },
    handleRowButtonClick() {
      this.$emit('rowButtonClick', this.row)
    },
  },
  render(h) {
    //有其他类型可以继续添加
    const { type, name, config } = this.header,
      row = this.row,
      data = this.row[name]

    const TYPE_COMP = {
      tag: ({ data, config, row }) => h(ctag, { props: { config, row, value: data, } }),
      operation: ({ config, row }) => h(OperationButton, {
        props: { btnList: config && config.btnList, type: 'icon', row, },
        on: { operBtnClick: this.handleOperBtnClick }
      }),
      map: ({ data, config }) => h('span', config.map[data]),
      def: ({ data, config, row }) => h('span', { style: this.getRowColor(config, row) }, data),
    }

    const VNodes = TYPE_COMP[type || 'def']({ data, config, row })
    return VNodes
  },
}
