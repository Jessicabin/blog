# 基于 echarts 的组件封装

由于 echarts 在使用时都需要初始化，以及注册 resize 事件，每个文件都要写这些代码，所以封装 echarts 组件更方便调用。

从外部传入echart 配置并在组件外添加宽高样式，即可展示统计图表。

echarts 图表展示一定需要限定 div 宽高，否则不会显示。

# echart 按需引入

echarts 依赖库比较大，避免占用过多空间，使用按需引入是比较好的选择。

// 引入 echarts 核心模块，核心模块提供了 echarts 使用必须要的接口。
import * as echarts from "echarts/core";
// 引入柱状图图表，图表后缀都为 Chart
import { BarChart, PieChart, LineChart } from "echarts/charts";
// 引入提示框，标题，直角坐标系，数据集，内置数据转换器组件，组件后缀都为 Component
import {
  TitleComponent,
  LegendComponent,
  TooltipComponent,
  GridComponent,
  DatasetComponent,
  TransformComponent,
} from "echarts/components";
// 标签自动布局，全局过渡动画等特性
import { LabelLayout, UniversalTransition } from "echarts/features";
// 引入 Canvas 渲染器，注意引入 CanvasRenderer 或者 SVGRenderer 是必须的一步
import { CanvasRenderer } from "echarts/renderers";

// 注册必须的组件
echarts.use([
  TitleComponent,
  TooltipComponent,
  GridComponent,
  DatasetComponent,
  TransformComponent,
  LegendComponent,
  BarChart,
  PieChart,
  LineChart,
  LabelLayout,
  UniversalTransition,
  CanvasRenderer,
]);

export default echarts;

如果打包方式使用 vite，可以在 vite.config.js 中加入以下代码，分块打包，减小 index 文件体积

export default defineConfig({
  build: {
    rollupOptions: {
      output: {
        //打包优化，分块打包，减小index文件体积
        manualChunks: {
          echarts: ["echarts"],
        },
      },
    },
  },
});

# 封装组件

echarts-box.vue

展示echarts的div使用随机id。

为了兼容一些空数据情况，无配置或无接口数据时，显示暂无数据。

外层再包裹一层div，默认宽高均为100%，使用时候需要加上具体的宽高样式。

<template>
  <div class="module-box" v-loading="loading" element-loading-text="正在加载中">
    <div v-if="!loading && !showChart" class="no-data abs">暂无数据</div>
    <div :id="id" class="chart-box" />
  </div>
</template>

import echarts from "./echarts";
import { debounce } from "lodash";

export default {
  props: {
    loading: {
      default: false,
    },
  },
  data() {
    return {
      chart: null,
      showChart: true,
    };
  },
  methods: {
    initChart(option) {
      if (option) {
        this.showChart = true;
        this.$nextTick(() => {
          if (this.chart) this.chart.setOption(option, true);
          else {
            this.chart = echarts.init(document.getElementById(this.id));
            this.chart.setOption(option, true);
          }
        });
      } else {
        this.showChart = false;
      }
    },
    resizeHandler() {
      return debounce(() => {
        if (this.chart) {
          this.chart.resize();
        }
      }, 100)();
    },
    initResizeEvent() {
      window.addEventListener("resize", this.resizeHandler);
    },
    destroyResizeEvent() {
      window.removeEventListener("resize", this.resizeHandler);
    },
  },
  computed: {
    id() {
      return "chart" + new Date().getTime() + Math.round(Math.random() * 100);
    },
  },
  mounted() {
    this.initResizeEvent();

    this.showChart = true;
  },
  beforeDestroy() {
    this.destroyResizeEvent();

    // 销毁echarts实例
    if (!this.chart) {
      return;
    }
    this.chart.dispose();
    this.chart = null;
  },
  activated() {
    this.initResizeEvent();
  },
  deactivated() {
    this.destroyResizeEvent();
  },
};

.module-box {
  position: relative;
  height: 100%;
  width: 100%;
}

.chart-box {
  position: relative;
  height: inherit;
  width: 100%;
}

.no-data {
  font-size: 14px;
  font-weight: 400;
  color: #fff;
  line-height: 20px;
  height: 20px;
  width: max-content;
}

# 使用示例

调用组件的initChart方法，传入echarts配置，即可展示echarts图表。

<echarts-box ref="echart" :loading="loading"></echarts-box>

const option = {} //echarts配置
this.$refs.echart.initChart(option)
